   193→// ============================================
   194→// Compact VND formatter for charts
   195→// ============================================
   196→function compactVND(value: number): string {
   197→  if (value >= 1_000_000_000) return `${(value / 1_000_000_000).toFixed(1)}B`;
   198→  if (value >= 1_000_000) return `${(value / 1_000_000).toFixed(1)}M`;
   199→  if (value >= 1_000) return `${(value / 1_000).toFixed(0)}K`;
   200→  return String(value);
   201→}

   232→export default function ReportsPage() {
   233→  const { locale } = useLocale();
   234→  const t = (en: string, vi: string) => locale === 'vi' ? vi : en;
   235→
   236→  const [period, setPeriod] = useState('all');
   237→  const [activeTab, setActiveTab] = useState('overview');
   238→
   239→  // Data state
   240→  const [overview, setOverview] = useState<OverviewData | null>(null);
   241→  const [revenue, setRevenue] = useState<RevenueData | null>(null);
   242→  const [ordersData, setOrdersData] = useState<OrdersData | null>(null);
   243→  const [productsData, setProductsData] = useState<ProductsData | null>(null);
   244→  const [shopsData, setShopsData] = useState<ShopsData | null>(null);
   245→  const [shipmentsData, setShipmentsData] = useState<ShipmentsData | null>(null);
   246→
   247→  const [loading, setLoading] = useState(false);
   248→  const [overviewLoading, setOverviewLoading] = useState(true);
   249→
   250→  // Fetch overview
   251→  const fetchOverview = useCallback(async () => {
   252→    try {
   253→      setOverviewLoading(true);
   254→      const res = await adminFetch(`/api/reports/overview?period=${period}`);
   255→      const json = await res.json();
   256→      if (json.success) setOverview(json.data);
   257→    } catch (err) { console.error('Overview fetch error:', err); }
   258→    finally { setOverviewLoading(false); }
   259→  }, [period]);
   260→
   261→  // Fetch tab-specific data
   262→  const fetchTabData = useCallback(async (tab: string) => {
   263→    if (tab === 'overview') return;
   264→    setLoading(true);
   265→    try {
   266→      const res = await adminFetch(`/api/reports/${tab === 'revenue' ? 'revenue' : tab === 'orders' ? 'orders' : tab === 'products' ? 'products' : tab === 'shops' ? 'shops-analytics' : tab === 'shipments' ? 'shipments-analytics' : ''}?period=${period}`);
   267→      const json = await res.json();
   268→      if (json.success) {
   269→        switch (tab) {
   270→          case 'revenue': setRevenue(json.data); break;
   271→          case 'orders': setOrdersData(json.data); break;
   272→          case 'products': setProductsData(json.data); break;
   273→          case 'shops': setShopsData(json.data); break;
   274→          case 'shipments': setShipmentsData(json.data); break;
   275→        }
   276→      }
   277→    } catch (err) { console.error(`${tab} fetch error:`, err); }
   278→    finally { setLoading(false); }
   279→  }, [period]);
   280→
   281→  useEffect(() => { fetchOverview(); }, [fetchOverview]);
   282→  useEffect(() => { fetchTabData(activeTab); }, [activeTab, fetchTabData]);
   283→
   284→  const handlePeriodChange = (newPeriod: string) => {
   285→    setPeriod(newPeriod);
   286→    setActiveTab('overview');
   287→    setRevenue(null); setOrdersData(null); setProductsData(null);
   288→    setShopsData(null); setShipmentsData(null);
   289→  };
   290→
   291→  const handleRefresh = () => {
   292→    fetchOverview();
   293→    fetchTabData(activeTab);
   294→  };
   295→
   296→  // Export CSV for current tab
   297→  const handleExport = () => {
   298→    // Build CSV based on active tab
   299→    let csv = '';
   300→    const BOM = '\uFEFF';
   301→
   302→    if (activeTab === 'overview' && overview) {
   303→      csv = 'KPI,Value\n';
   304→      csv += `${t('Total Revenue', 'Tong Doanh thu')},${overview.kpis.totalRevenue}\n`;
   305→      csv += `${t('Total Orders', 'Tong Don hang')},${overview.kpis.totalOrders}\n`;
   306→      csv += `${t('Active Shops', 'Cua hang HD')},${overview.kpis.activeShops}\n`;
   307→      csv += `${t('Success Rate', 'Ti le thanh cong')},${overview.kpis.successRate}%\n`;
   308→    } else if (activeTab === 'orders' && ordersData) {
   309→      csv = 'Date,Orders,Completed,Cancelled,Revenue\n';
   310→      ordersData.trends.daily.forEach(d => {
   311→        csv += `${d.date},${d.orders},${d.completed},${d.cancelled},${d.revenue}\n`;
   312→      });
   313→    } else if (activeTab === 'revenue' && revenue) {
   314→      csv = 'Date,Revenue,Orders,AOV\n';
   315→      revenue.trends.daily.forEach(d => {
   316→        csv += `${d.date},${d.revenue},${d.orders},${d.avgOrderValue}\n`;
   317→      });
   318→    } else if (activeTab === 'products' && productsData) {
   319→      csv = 'Product,SKU,Revenue,Qty,Category,Brand\n';
   320→      productsData.rankings.topProducts.forEach(p => {
   321→        csv += `"${p.name}","${p.sku}",${p.revenue},${p.qty},"${p.category}","${p.brand}"\n`;
   322→      });
   323→    } else if (activeTab === 'shops' && shopsData) {
   324→      csv = 'Name,District,Tier,Orders,GMV\n';
   325→      shopsData.rankings.topByGmv.forEach(s => {
   326→        csv += `"${s.name}","${s.district || ''}","${s.loyaltyTier}",${s.totalOrders},${s.totalGmv}\n`;
   327→      });
   328→    } else if (activeTab === 'shipments' && shipmentsData) {
   329→      csv = 'Date,Total,Delivered,Failed,InTransit\n';
   330→      shipmentsData.trends.daily.forEach(d => {
   331→        csv += `${d.date},${d.total},${d.delivered},${d.failed},${d.inTransit}\n`;
   332→      });
   333→    }
   334→
   335→    if (!csv) return;
   336→    const blob = new Blob([BOM + csv], { type: 'text/csv;charset=utf-8;' });
   337→    const url = URL.createObjectURL(blob);
   338→    const a = document.createElement('a');
   339→    a.href = url;
   340→    a.download = `aladin-report-${activeTab}-${period}-${new Date().toISOString().split('T')[0]}.csv`;
   341→    a.click();
   342→    URL.revokeObjectURL(url);
   343→  };
   344→
   345→  // ============================================
   346→  // OVERVIEW TAB
   347→  // ============================================
   348→  const renderOverview = () => {
   349→    if (overviewLoading) {
   350→      return (
   351→        <div className="space-y-4">
   352→          <div className="grid grid-cols-2 sm:grid-cols-4 gap-3">
   353→            {Array.from({ length: 8 }).map((_, i) => <Skeleton key={i} className="h-24 w-full rounded-lg" />)}
   354→          </div>
   355→          <Skeleton className="h-64 w-full rounded-lg" />
   356→        </div>
   357→      );
   358→    }
   359→
   360→    if (!overview) {
   361→      return (
   362→        <div className="text-center py-12">
   363→          <BarChart3 className="h-12 w-12 mx-auto text-muted-foreground mb-3" />
   364→          <p className="text-sm text-muted-foreground">{t('No data available', 'Khong co du lieu')}</p>
   365→        </div>
   366→      );
   367→    }
   368→
   369→    const k = overview.kpis;
   370→
   371→    return (
   372→      <div className="space-y-4">
   373→        {/* KPI Grid */}
   374→        <div className="grid grid-cols-2 sm:grid-cols-4 gap-3">
   375→          <KpiCard title="Revenue" titleVi="Doanh thu" value={k.totalRevenueFormatted} icon={<DollarSign className="h-4 w-4" />} growth={k.revenueGrowth} locale={locale} variant="success" />
   376→          <KpiCard title="Orders" titleVi="Don hang" value={k.totalOrders} icon={<ShoppingCart className="h-4 w-4" />} growth={k.orderGrowth} locale={locale} />
   377→          <KpiCard title="Active Shops" titleVi="Cua hang HD" value={k.activeShops} icon={<Store className="h-4 w-4" />} locale={locale} suffix={`/ ${k.totalShops}`} />
   378→          <KpiCard title="Avg Order Value" titleVi="TB Don hang" value={k.avgOrderValueFormatted} icon={<Package className="h-4 w-4" />} locale={locale} />
   379→          <KpiCard title="Shipments" titleVi="Van chuyen" value={k.deliveredShipments} icon={<Truck className="h-4 w-4" />} locale={locale} suffix={`/ ${k.totalShipments}`} growth={k.successRateDelta ? k.successRateDelta : undefined} />
   380→          <KpiCard title="Success Rate" titleVi="TL Thanh cong" value={`${k.successRate}%`} icon={<CheckCircle2 className="h-4 w-4" />} locale={locale} variant={k.successRate >= 90 ? 'success' : k.successRate >= 70 ? 'warning' : 'danger'} />
   381→          <KpiCard title="Credit Exposure" titleVi="Cong no" value={k.totalCreditExposureFormatted} icon={<CreditCard className="h-4 w-4" />} locale={locale} variant={k.overdueShops > 0 ? 'warning' : 'default'} />
   382→          <KpiCard title="Group Buy Savings" titleVi="Tiet kiem MC" value={k.totalSavingsFormatted} icon={<Tag className="h-4 w-4" />} locale={locale} variant="info" />
   383→        </div>
   384→
   385→        {/* Revenue Trend + Distributions Row */}
   386→        <div className="grid grid-cols-1 lg:grid-cols-3 gap-4">
   387→          <div className="lg:col-span-2">
   388→            <BarChart
   389→              title="Revenue Trend" titleVi="Xu huong Doanh thu"
   390→              data={overview.dailyRevenue.map(d => ({
   391→                label: d.date.slice(5),
   392→                value: d.revenue,
   393→                color: d.date === overview.dailyRevenue[overview.dailyRevenue.length - 1]?.date ? 'bg-red-500' : 'bg-red-500/60',
   394→              }))}
   395→              formatValue={compactVND}
   396→              locale={locale}
   397→              height={200}
   398→            />
   399→          </div>
   400→
   401→          <DistributionChart
   402→            title="Order Status" titleVi="Trang thai Don hang"
   403→            data={Object.entries(overview.distributions.orderStatus)

   413→            locale={locale}
   414→          />
   415→        </div>
   416→
   417→        {/* Bottom Row: Top Categories + Payment + Tier */}
   418→        <div className="grid grid-cols-1 md:grid-cols-3 gap-4">
   419→          <HBarChart
   420→            title="Top Categories" titleVi="Danh muc hang dau"
   421→            data={overview.topCategories.slice(0, 8).map(c => ({
   422→              label: c.name, value: c.revenue,
   423→              subtitle: c.revenueFormatted,
   424→            }))}
   425→            formatValue={compactVND}
   426→            locale={locale}
   427→            maxItems={8}
   428→          />
   429→
   430→          <DistributionChart
   431→            title="Payment Method" titleVi="Phuong thuc TT"
   432→            data={Object.entries(overview.distributions.paymentMethod)
   433→              .sort((a, b) => b[1] - a[1])
   434→              .map(([method, value]) => ({
   435→                label: PAYMENT_METHOD_LABELS[method]?.en || method,
   436→                labelVi: PAYMENT_METHOD_LABELS[method]?.vi || method,
   437→                value,
   438→                color: method === 'CREDIT' ? 'bg-amber-500' : method === 'DIGITAL' ? 'bg-blue-500' : 'bg-red-500',
